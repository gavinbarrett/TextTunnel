Users Table
-----------
create table users (username varchar(64), password varchar(64), email varchar(64), created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP);

-- Grant privileges --
grant all privileges on table users to gavin;


Friendship Table
----------------
create table friendships (friend1 varchar(64), friend2 varchar(64), status varchar(8) default 'None');

-- Grant privileges --
grant all privileges on table friendships to gavin;


Channel Table
-------------

create table channels (channelname varchar(64), accesslevel varchar(7), accessmode varchar(3), credentials varchar(64), created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP);

-- Grant privileges --
grant all privileges on table channels to gavin;
